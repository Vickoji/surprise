const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); const yesBtn = document.getElementById('yesBtn'); // Needed for reference const noBtn = document.getElementById('noBtn'); const container = document.getElementById('mainContainer'); let particles = []; let currentText = "Vicks"; function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; } window.addEventListener('resize', resize); resize(); class Particle { constructor() { this.init(); } init() { this.x = Math.random() * canvas.width; this.y = canvas.height + Math.random() * 200; this.text = currentText; this.size = Math.random() * 20 + 15; this.speed = Math.random() * 2 + 1; this.opacity = Math.random() * 0.6 + 0.2; this.angle = Math.random() * Math.PI * 2; this.spin = Math.random() * 0.02 - 0.01; } draw() { ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.angle); ctx.fillStyle = `rgba(214, 51, 132, ${this.opacity})`; ctx.font = `bold ${this.size}px 'Quicksand'`; ctx.fillText(this.text, 0, 0); ctx.font = `${this.size/1.5}px Arial`; ctx.fillText("❤️", this.size * 1.5, 0); ctx.restore(); } update() { this.y -= this.speed; this.angle += this.spin; this.x += Math.sin(this.y * 0.01); if (this.y < -50) this.init(); } } for (let i = 0; i < 40; i++) particles.push(new Particle()); function animate() { ctx.clearRect(0, 0, canvas.width, canvas.height); particles.forEach(p => { p.update(); p.draw(); }); requestAnimationFrame(animate); } animate(); // --- NEW CONSTRAINED MOVING LOGIC --- noBtn.addEventListener('mouseover', () => { const radius = 380; noBtn.style.top = newY + 'px'; }); function celebrate() { currentText = "Nicks"; particles = []; for (let i = 0; i < 100; i++) { setTimeout(() => { particles.push(new Particle()); }, i * 20); } container.style.transform = "scale(0) rotate(720deg)";
